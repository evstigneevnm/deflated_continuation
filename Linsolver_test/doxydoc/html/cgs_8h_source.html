96  vec_ops_->add_mul(T(1.f), b, -T(1.f), ri); //ri := -ri + b = -A*x0 + b

98  T rho_i_2;
99  bool not_valid_coeff_faced = false;
100 
101  while (!monitor_.check_finished(x, ri))
102  {
103  if (prec_ != NULL) prec_->apply(ri); //ri := P*ri
104 
105  if (monitor_.iters_performed() == 0) {
106  //r0 := ri;
107  vec_ops_->assign(ri, r0);
108  }
109 
110  T rho_i_1 = vec_ops_->scalar_prod(ri, r0);
111  //TODO check whether rho_k is zero
112  if (monitor_.iters_performed() == 0) {
113  vec_ops_->assign(ri, ui);
114  vec_ops_->assign(ui, pi);
115  } else {
116  T beta_i_1 = rho_i_1/rho_i_2;
117  if (isnan(beta_i_1)) { logged_obj_t::info("solve: stop iterations because beta_i_1 is nan"); not_valid_coeff_faced = true; break; }
118  if (isinf(beta_i_1)) { logged_obj_t::info("solve: stop iterations because beta_i_1 is inf"); not_valid_coeff_faced = true; break; }
119  //ui := ri + beta_i_1 * qi
120  vec_ops_->assign_mul(T(1.f), ri, beta_i_1, qi, ui);
121  //pi := beta_i_1*beta_i_1 * p{i-1} + ui + beta_i_1 * qi
122  vec_ops_->add_mul( T(1.f), ui, beta_i_1, qi, beta_i_1*beta_i_1, pi);
123  }
124  //theta := P*A*pi
125  A.apply(pi, theta);
126  if (prec_ != NULL) prec_->apply(theta);
127  T alpha_i = rho_i_1/vec_ops_->scalar_prod(theta, r0);
128  if (isnan(alpha_i)) { logged_obj_t::info("solve: stop iterations because alpha_i is nan"); not_valid_coeff_faced = true; break; }
129  if (isinf(alpha_i)) { logged_obj_t::info("solve: stop iterations because alpha_i is inf"); not_valid_coeff_faced = true; break; }
130  //qi := ui - alpha_i*theta
131  vec_ops_->assign_mul(T(1.f), ui, -alpha_i, theta, qi);
132  //theta := ui + qi;
133  vec_ops_->assign_mul(T(1.f), ui, T(1.f), qi, theta);
134  //xi := x{i-1} + theta*alpha_i
135  vec_ops_->add_mul(alpha_i, theta, T(1.f), x);
136 
137  //ri := b - A*xi;
138  A.apply(x, ri); //ri := A*xi
139  vec_ops_->add_mul(T(1.f), b, -T(1.f), ri); //ri := -ri + b = -A*xi + b
140

142

144  }

147 
148  if (not_valid_coeff_faced)
149  return true;
150  else
151  return monitor_.converged();
152  }
