112  virtual bool solve(const linear_operator_type &A, const vector_type &b,
113  vector_type &x)const
114  {
115  if ((!use_precond_resid_)&&(prec_ != NULL))
116  throw std::logic_error("bicgstabl::solve: use_precond_resid_ == false with non-empty preconditioner is not supported");
117  if (prec_ != NULL) prec_->set_operator(&A);

120  use_wrap_r.start_use_all(); use_wrap_u.start_use_all();

125  monitor_call_wrap_t monitor_wrap(monitor_);
126  if ((use_precond_resid_)&&(prec_ != NULL)) {
127  vec_ops_->assign(b, r[0]);
128  prec_->apply(r[0]);
129  monitor_wrap.start(r[0]);
130  } else {
131  monitor_wrap.start(b);
132  }
133 
134  vector_type *checked_ri;
135  if (use_real_resid_) {
136  checked_ri = &real_ri;
137  } else {
138  checked_ri = &r[0];
139  }
140 
141  vec_ops_->assign_scalar(T(0.f), u[0]); //u[0] := 0.;
142 
143  T tau[basis_sz_][basis_sz_]; //Hessenberg matrix
144  T sigma[basis_sz_ + 1];
145  T gamma[basis_sz_ + 1];
146  T gamma_p[basis_sz_ + 1];
147  T gamma_pp[basis_sz_ + 1];
148 
149  calc_residual_(A, x, b, r[0]);
150  vec_ops_->assign_mul(T(1.f), r[0], rtilde);
151  if (use_real_resid_) vec_ops_->assign(r[0], real_ri);
152  normalize_(rtilde);
153 
154  T rho = T(1.f), alpha = T(0.f), omega = T(1.f);
155 
156  bool res = true;
157 
158  flag = 1; //default termiantion flag
159 
160  while (!monitor_.check_finished(x, *checked_ri))
161  {
162  rho = -omega * rho;
163  for (int j = 0; j < basis_sz_; ++j) { //j=0,...basis_sz_-1
164  if ( rho == T(0.f) ) { //check rho break
165  flag=-1;
166  break;
167  }
168 
169  T rho1 = vec_ops_->scalar_prod(r[j], rtilde); //rho1=(r[j],rtilde)
170  T beta = alpha * rho1 / rho;
171  rho = rho1;
172  for (int i = 0; i <= j; ++i){
173  vec_ops_->add_mul(T(1.f), r[i], -beta, u[i]); //u[i] := r[i] - beta * u[i]
174  }
175  A.apply(u[j], u[j+1]); //u[j+1] := A*u[j]
176  if (prec_ != NULL) prec_->apply(u[j+1]);
177  alpha = rho / vec_ops_->scalar_prod(u[j+1], rtilde); //alpha=rho/(u[j+1],rtilde)
178  for (int i = 0; i <= j; ++i){
179  vec_ops_->add_mul(-alpha, u[i+1], T(1.f), r[i]); //r[i] := r[i] - alpha * u[i+1]
180  }
181  A.apply(r[j], r[j+1]); //r[j+1] := A*r[j] Krylov subspace
182  if (prec_ != NULL) prec_->apply(r[j+1]);
183  vec_ops_->add_mul(alpha, u[0], T(1.f), x); //x := x + alpha * u[0]
184  }
185  for (int j = 1; j <= basis_sz_; ++j) {
186  for (int i = 1; i < j; ++i) {
187  tau[i][j] = vec_ops_->scalar_prod(r[j], r[i]) / sigma[i];
188  vec_ops_->add_mul(-tau[i][j], r[i], T(1.f), r[j]); //r[j] := r[j] - tau[i,j] * r[i]
189  }
190  sigma[j] = vec_ops_->scalar_prod(r[j],r[j]); //sigma[j]=(r[j],r[j]);
191  gamma_p[j] = vec_ops_->scalar_prod(r[0], r[j]) / sigma[j]; //gamma_p[j]=(r[0],r[j])/sigma[j];
192  }
193  gamma[basis_sz_] = gamma_p[basis_sz_]; //gamma[basis_sz_]=gamma_p[basis_sz_];
194  omega = gamma[basis_sz_]; //gamma=gamma[basis_sz_];
195  for (int j = basis_sz_-1; j >= 1; --j) {
196  gamma[j] = gamma_p[j]; //gamma[j]=gamma_p[j]
197  for (int i = j+1; i <= basis_sz_; ++i) {
198  gamma[j] -= tau[j][i] * gamma[i]; //gamma[j]=gamma[j]-tau[j,i].*gamma[i];
199  }
200  }
201  for (int j = 1; j < basis_sz_; ++j) {
202  gamma_pp[j] = gamma[j+1]; //gamma_pp[j]=gamma[j+1]
203  for (int i = j+1; i < basis_sz_; ++i){
204  gamma_pp[j] += tau[j][i] * gamma[i+1]; //gamma_pp[j]=gamma_pp[j]+tau[j,i]*gamma[i+1]
205  }
206  }
207  vec_ops_->add_mul(gamma[1], r[0], T(1.f), x); //x := x + gamma[1] * r[0];
208  vec_ops_->add_mul(-gamma_p[basis_sz_], r[basis_sz_], T(1.f), r[0]); //r[0] := r[0] - gamma_p[basis_sz_] * r[basis_sz_]
209  vec_ops_->add_mul(-gamma[basis_sz_], u[basis_sz_], T(1.f), u[0]); //u[0] := u[0] - gamma[basis_sz_] * u[basis_sz_];
210 
211  for (int j = 1; j < basis_sz_; ++j) { //j=1,..basis_sz_-1
212  vec_ops_->add_mul(gamma_pp[j], r[j], T(1.f), x); //x := x + gamma_pp[j] * r[j]
213  vec_ops_->add_mul(-gamma_p[j], r[j], T(1.f), r[0]); //r[0] := r[0] - gamma_p[j] * r[j]
214  vec_ops_->add_mul(-gamma[j], u[j], T(1.f), u[0]); //u[0] := u[0] - gamma[j] * u[j]
215  }
216 
217  if (use_real_resid_) calc_residual_(A, x, b, real_ri);
218 
219  monitor_ += basis_sz_;
220  }
221 
222  return res;
