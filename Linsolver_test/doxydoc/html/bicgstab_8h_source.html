109  virtual bool solve(const linear_operator_type &A, const vector_type &b,
110  vector_type &x)const
111  {
112  if (prec_ != NULL){
113  prec_->set_operator(&A);
114  }
115 
116  bufs_arr_use_wrap_t use_wrap(bufs);
117  use_wrap.start_use_all();
118 
119  monitor_call_wrap_t monitor_wrap(monitor_);
120  if (use_precond_resid_) {
121  vec_ops_->assign(b, ri);
122  if (prec_ != NULL) prec_->apply(ri);
123  monitor_wrap.start(ri);
124  } else {
125  monitor_wrap.start(b);
126  }
127 
128  vector_type *checked_ri;
129  if (use_precond_resid_) {
130  checked_ri = &ri;
131  } else {
132  checked_ri = &nonprecond_ri;
133  }
134 
135  //ri := P*b - P*A*x0;
136  A.apply(x, ri); //ri := A*x0
137  vec_ops_->add_mul(T(1), b, -T(1), ri); //ri := -ri + b = -A*x0 + b
138  if (!use_precond_resid_) vec_ops_->assign(ri, nonprecond_ri);
139  if (prec_ != NULL) prec_->apply(ri); //ri := P*ri = -P*A*x0 + P*b
140  //r_ := ri;
141  vec_ops_->assign(ri, r_);
142 
143  T rho_i_1(1), alpha(1), omega_i_1(1);
144 
145  vec_ops_->assign_scalar(T(0), nu_i);
146  vec_ops_->assign_scalar(T(0), pi);
147 
148  bool not_valid_coeff_faced = false;
149 
150  while (!monitor_.check_finished(x, *checked_ri))
151  {
152  //nu_i, pi, ri are nu_{i-1}, p_{i-1}, r_{i-1}
153  T rho_i = vec_ops_->scalar_prod(ri, r_),
154  beta = (rho_i/rho_i_1)*(alpha/omega_i_1);
155  if (isnan(beta)) { logged_obj_t::info("solve: stop iterations because beta is ind"); not_valid_coeff_faced = true; break; }
156  if (isinf(beta)) { logged_obj_t::info("solve: stop iterations because beta is inf"); not_valid_coeff_faced = true; break; }
157 
158  //pi := ri - beta*omega_{i_1}*nu_i + beta*p_{i-1}
159  vec_ops_->add_mul(T(1), ri, -beta*omega_i_1, nu_i, beta, pi);
160  //pi now is pi
161  //nu_i := P*A*pi
162  A.apply(pi, nu_i);
163  if (!use_precond_resid_) {
164  //use t as buffer 4 nonprecond nu_i
165  vec_ops_->assign(nu_i, t);
166  }
167  if (prec_ != NULL) prec_->apply(nu_i);
168  //nu_i now is nu_i
169 
170  alpha = rho_i/vec_ops_->scalar_prod(nu_i, r_);
171  if (isnan(alpha)) { logged_obj_t::info("solve: stop iterations because alpha is nan"); not_valid_coeff_faced = true; break; }
172  if (isinf(alpha)) { logged_obj_t::info("solve: stop iterations because alpha is inf"); not_valid_coeff_faced = true; break; }
173 
174  //s := ri - alpha*nu_i
175  vec_ops_->assign_mul(T(1), ri, -alpha, nu_i, s);
176  if (!use_precond_resid_) {
177  //NOTE t here is nonpreconditioned nu_i
178  vec_ops_->add_mul(-alpha, t, T(1), nonprecond_ri);
179  //nonprecond_ri here is nonpreconditioned s
180  }
181 
182  //theta := P*A*pi
183  A.apply(s, t);
184  if (!use_precond_resid_) {
185  //use ri as buffer 4 nonprecond t
186  vec_ops_->assign(t, ri);
187  }
188  if (prec_ != NULL) prec_->apply(t);
189 
190  T omega_i = vec_ops_->scalar_prod(t, s)/vec_ops_->scalar_prod(t, t);
191  if (isnan(omega_i)) { logged_obj_t::info("solve: stop iterations because omega_i is nan"); not_valid_coeff_faced = true; break; }
192  if (isinf(omega_i)) { logged_obj_t::info("solve: stop iterations because omega_i is inf"); not_valid_coeff_faced = true; break; }
193 
194  //x := x + alpha*pi + omega_i*s
195  vec_ops_->add_mul(alpha, pi, omega_i, s, T(1), x);
196 
197  if ((resid_recalc_freq_ == 0)||((monitor_.iters_performed()+1) % resid_recalc_freq_ != 0)) {
198  if (!use_precond_resid_) {
199  //NOTE ri here is nonpreconditioned t
200  //NOTE nonprecond_ri here is nonpreconditioned s
201  vec_ops_->add_mul(-omega_i, ri, T(1), nonprecond_ri);
202  //nonprecond_ri now is nonpreconditioned ri
203  }
204 
205  //ri := s - omega_i*t
206  vec_ops_->assign_mul(T(1), s, -omega_i, t, ri);
207  //ri now is ri
208  } else {
209  logged_obj_t::info_f("solve: iter = %d recalc real residual", monitor_.iters_performed());
210 
211  //ri := P*b - P*A*x0;
212  A.apply(x, ri); //ri := A*x
213  vec_ops_->add_mul(T(1), b, -T(1), ri); //ri := -ri + b = -A*x + b
214  if (!use_precond_resid_) vec_ops_->assign(ri, nonprecond_ri);
215  if (prec_ != NULL) prec_->apply(ri); //ri := P*ri = -P*A*x + P*b
216  }
217 
218  omega_i_1 = omega_i; rho_i_1 = rho_i;
219  ++monitor_;
220  }
221 
222  if (monitor_.out_min_resid_norm()) vec_ops_->assign(monitor_.min_resid_norm_x(), x);
223 
224  if (not_valid_coeff_faced)
225  return true;
226  else
227  return monitor_.converged();
